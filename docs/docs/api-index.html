Quick API Reference

Create a Game

import { Game, Scene } from '@mode-7/zap';
const game = new Game({ parent: '#app', responsive: true });
const scene = new Scene({ backgroundColor: '#1a1a2e' });
game.setScene(scene);
game.start();

Add a Sprite

const sprite = new Sprite({
  x: 100, y: 100,
  width: 50, height: 50,
  color: '#ff0000'
});
scene.add(sprite);

Handle Tap

sprite.interactive = true;
sprite.on('tap', () => console.log('Tapped!'));

Animate

sprite.tween(
  { x: 200, rotation: Math.PI },
  { duration: 1000, easing: 'easeInOutQuad' }
);

Detect Collisions

sprite.checkCollisions = true;
sprite.collisionTags = ['enemy'];
sprite.on('collisionenter', (event) => {
  console.log('Hit:', event.other);
});
